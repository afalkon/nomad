mixin notifications(title, notifications)
    //- Проверяем тип нотификации и определяем нужный класс
    if (notifications == "error")
        - var typeNotifications = "notifications__title--error"
    else if  (notifications == "success")
        - var typeNotifications = "notifications__title--success"
    else
        - var typeNotifications = ""

        - var additionalClass = "notifications__title--with-message"
    else
        - var additionalClass = ""
    //- Создаем разметку
    .notifications(class=additionalClass)
        .notifications__title(class= typeNotifications) #{title}

        if block
